* パッケージ管理
** レポジトリを登録する
#+begin_src emacs-lisp
  (require 'package)
  (setq package-archives
	'(("melpa" . "https://melpa.org/packages/")
	  ;; ("melpa-stable" . "https://stable.melpa.org/packages/")
	  ("org" . "https://orgmode.org/elpa/")
	  ("gnu" . "https://elpa.gnu.org/packages/")))
  (package-initialize)
#+end_src

    :ensure t
    :config

     'org-babel-load-languages
